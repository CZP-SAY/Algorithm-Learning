```c++
#include <bits/stdc++.h>
using namespace std;
const int N = 1e5 + 5;
#define long long int
void solve()
{
     int n;
     cin >> n;
     vector<int> ans(n);
     //对于每一个a,找一个b(b>a,b<=n-1),使得a+b符合条件,然后a~b的所有的数就都有了答案
     for (int i = n - 1, j; i >= 0; i = j - 1)
     {
          for (j = i; j >= 0; j--)
          {
               int t = (i + j);
               int k = sqrt(t);
               if (k * k == t) //找到了平方数
                    break;
          }

          for (int k = j; k <= i; k++)
          {
               ans[k] = i - (k - j);
          }
     }

     for (int i = 0; i < ans.size(); i++)
          cout << ans[i] << ' ';

     cout << "\n";
}
signed main()
{
     int t;
     cin >> t;
     while (t--)
     {
          solve();
     }
}
```
